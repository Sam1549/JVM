    public static void main(String[] args) {
    // В стеке сохдается фрейм main
        int i = 1;                      // 1 - Во фрейме main создается переменная i - со значением 1.
        Object o = new Object();        // 2 - Во фрейме main создается переменная o - которая хранит ссылку на обьект, а сам обьект создается в heap(куче)

        printAll(o, i, ii);             // 4 - В стеке создается фрейм printAll, в нем создаеются два обьекта - ссылающиеся на уже созданые обьекты в heap(куче),

        System.out.println("finished"); // 7 - В стеке создается фрейм println, в котором создается переменная ссылающиеся на обьект типа String. 
    }
    // После выхода из метода main - из стека удаляется фрейм main

    private static void printAll(Object o, int i, Integer ii) {
        Integer uselessVar = 700;                   // 5 - Во фрейме printAll создается переменная uselessVar - которая хранит в себе ссылку на обьект типа Integer,
                                                    //     созданый в heap(куче) со значением 700.
        System.out.println(o.toString() + i + ii);  // 6 - В стеке создаются фреймы println и toString.
                                                    //     В стеке println создаются два обьекта o и ii - ссылающиеся на уже созданые обьекты в heap(куче)
                                                    //     также создается переменная i со значением 1
    }
    // Когда выйдем из метода printAll - из стека удаляется фрейм printAll
}
